def []=(key, value)
  raise ArgumentError, "#{value} is not a Tube" unless value.is_a? Tube or value.nil?

  case key
  when /\A([a-zA-Z])(\d+)\z/
    position = element_name_to_index($1, $2)
    raise RackPositionNotEmpty unless content[position].nil? or value.nil?
    content[position] = value
  when Symbol
    self[key.to_s] = value
  else
    super(key, value)
  end
end
