def [](index)
	case index
	when Array
		get_element(*index)
	when /\A([a-zA-Z])(\d+)\z/
		get_element($1, $2)
		# why not something like
		# get_well(*[$1, $2].zip(%w{A 1}).map { |a,b| [a,b].map(&:ord).inject { |x,y| x-y } })
	when Symbol
		self[index.to_s]
	else
		super(index)
	end
end
