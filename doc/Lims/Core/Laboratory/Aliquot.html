def take_fraction(fraction)
  new = self.class.new(attributes)
  if quantity && fraction
    new_quantity = quantity*fraction
    self.quantity -= new_quantity
    new.quantity = new_quantity
  else
    new.quantity= nil
  end
  return new
end
