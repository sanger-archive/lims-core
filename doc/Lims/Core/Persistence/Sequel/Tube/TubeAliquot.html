def load_aliquots(tube_id)
  dataset.join(Aliquot::dataset(@session), :id => :aliquot_id).filter(:tube_id => tube_id).each do |att|
    att.delete(:id)
    aliquot  = @session.aliquot.get_or_create_single_model(att[:aliquot_id], att)
    yield(aliquot)
  end
end
